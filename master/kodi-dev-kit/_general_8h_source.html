748inline bool ATTRIBUTE_HIDDEN GetKeyboardLayout(int modifierKey,
749 std::string& layout_name,
750 std::vector<std::vector<std::string>>& layout)
751{
752 using namespace kodi::addon;
753
754 AddonToKodiFuncTable_Addon* toKodi = CAddonBase::m_interface->toKodi;
755 AddonKeyboardKeyTable c_layout;
756 char* c_layout_name = nullptr;
757 bool ret =
758 toKodi->kodi->get_keyboard_layout(toKodi->kodiBase, &c_layout_name, modifierKey, &c_layout);
759 if (ret)
760 {
761 if (c_layout_name)
762 {
763 layout_name = c_layout_name;
764 toKodi->free_string(toKodi->kodiBase, c_layout_name);
765 }
766
767 layout.resize(STD_KB_BUTTONS_MAX_ROWS);
768 for (unsigned int row = 0; row < STD_KB_BUTTONS_MAX_ROWS; row++)
769 {
770 layout[row].resize(STD_KB_BUTTONS_PER_ROW);
771 for (unsigned int column = 0; column < STD_KB_BUTTONS_PER_ROW; column++)
772 {
773 char* button = c_layout.keys[row][column];
774 if (button)
775 {
776 layout[row][column] = button;
777 toKodi->free_string(toKodi->kodiBase, button);
778 }
779 }
780 }
781 }
782 return ret;
783}
