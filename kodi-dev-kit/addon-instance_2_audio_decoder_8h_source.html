598 inline static std::string GetTrack(const std::string& name,
599 const std::string& trackPath,
600 int& track)
601 {
602 /*
603 * get the track name from path
604 */
605 track = 0;
606 std::string toLoad(trackPath);
607 const std::string ext = "." + name + KODI_ADDON_AUDIODECODER_TRACK_EXT;
608 if (toLoad.find(ext) != std::string::npos)
609 {
610 size_t iStart = toLoad.rfind('-') + 1;
611 track = atoi(toLoad.substr(iStart, toLoad.size() - iStart - ext.size()).c_str());
612 // The directory we are in, is the file
613 // that contains the bitstream to play,
614 // so extract it
615 size_t slash = trackPath.rfind('\\');
616 if (slash == std::string::npos)
617 slash = trackPath.rfind('/');
618 toLoad = trackPath.substr(0, slash);
619 }
620
621 return toLoad;
622 }

647 inline static bool ADDON_init(const KODI_ADDON_AUDIODECODER_HDL hdl,
648 const char* file,
649 unsigned int filecache,
650 int* channels,
651 int* samplerate,
652 int* bitspersample,
653 int64_t* totaltime,
654 int* bitrate,
655 AudioEngineDataFormat* format,
656 enum AudioEngineChannel info[AUDIOENGINE_CH_MAX])
657 {
658 CInstanceAudioDecoder* thisClass = static_cast<CInstanceAudioDecoder*>(hdl);
659
660 std::vector<AudioEngineChannel> channelList;
661
662 bool ret = thisClass->Init(file, filecache, *channels, *samplerate, *bitspersample, *totaltime,
663 *bitrate, *format, channelList);
664 if (!channelList.empty())
665 {
666 if (channelList.back() != AUDIOENGINE_CH_NULL)
667 channelList.push_back(AUDIOENGINE_CH_NULL);
668
669 for (unsigned int i = 0; i < channelList.size(); ++i)
670 {
671 info[i] = channelList[i];
672 }
673 }
674 return ret;
675 }

690 inline static bool ADDON_read_tag(const KODI_ADDON_AUDIODECODER_HDL hdl,
691 const char* file,
692 struct KODI_ADDON_AUDIODECODER_INFO_TAG* tag)
693 {
694#ifdef _WIN32
695#pragma warning(push)
696#pragma warning(disable : 4996)
697#endif // _WIN32
698 kodi::addon::AudioDecoderInfoTag cppTag;
699 bool ret = static_cast<CInstanceAudioDecoder*>(hdl)->ReadTag(file, cppTag);
700 if (ret)
701 {
702 tag->title = strdup(cppTag.GetTitle().c_str());
703 tag->artist = strdup(cppTag.GetArtist().c_str());
704 tag->album = strdup(cppTag.GetAlbum().c_str());
705 tag->album_artist = strdup(cppTag.GetAlbumArtist().c_str());
706 tag->media_type = strdup(cppTag.GetMediaType().c_str());
707 tag->genre = strdup(cppTag.GetGenre().c_str());
708 tag->duration = cppTag.GetDuration();
709 tag->track = cppTag.GetTrack();
710 tag->disc = cppTag.GetDisc();
711 tag->disc_subtitle = strdup(cppTag.GetDiscSubtitle().c_str());
712 tag->disc_total = cppTag.GetDiscTotal();
713 tag->release_date = strdup(cppTag.GetReleaseDate().c_str());
714 tag->lyrics = strdup(cppTag.GetLyrics().c_str());
715 tag->samplerate = cppTag.GetSamplerate();
716 tag->channels = cppTag.GetChannels();
717 tag->bitrate = cppTag.GetBitrate();
718 tag->comment = strdup(cppTag.GetComment().c_str());
719 std::string mimetype;
720 size_t size = 0;
721 const uint8_t* mem = cppTag.GetCoverArtByMem(size, mimetype);
722 if (mem && size > 0)
723 {
724 tag->cover_art_mem_mimetype = strdup(mimetype.c_str());
725 tag->cover_art_mem_size = size;
726 tag->cover_art_mem = static_cast<uint8_t*>(malloc(size));
727 memcpy(tag->cover_art_mem, mem, size);
728 }
729 else
730 {
731 tag->cover_art_path = strdup(cppTag.GetCoverArtByPath().c_str());
732 }
733 }
734 return ret;
735#ifdef _WIN32
736#pragma warning(pop)
737#endif // _WIN32
738 }
