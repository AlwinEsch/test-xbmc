712 void SetAddonStruct(KODI_HANDLE instance)
713 {
714 if (instance == nullptr)
715 throw std::logic_error("kodi::addon::CInstancePeripheral: Creation with empty addon structure not allowed, table must be given from Kodi!");
716
717 m_instanceData = static_cast<AddonInstance_Peripheral*>(instance);
718 m_instanceData->toAddon.addonInstance = this;
719
720 m_instanceData->toAddon.get_capabilities = ADDON_GetCapabilities;
721 m_instanceData->toAddon.perform_device_scan = ADDON_PerformDeviceScan;
722 m_instanceData->toAddon.free_scan_results = ADDON_FreeScanResults;
