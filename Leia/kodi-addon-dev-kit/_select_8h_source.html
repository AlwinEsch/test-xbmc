174 cEntries, size, selected, autoclose);
175 if (ret >= 0)
176 {
177 entries[ret].selected = true;

227 inline bool ShowMultiSelect(const std::string& heading, std::vector<SSelectionEntry>& entries, int autoclose = 0)
228 {
229 using namespace ::kodi::addon;
230 unsigned int size = entries.size();
231 const char** cEntryIDs = static_cast<const char**>(malloc(size*sizeof(const char*)));
232 const char** cEntryNames = static_cast<const char**>(malloc(size*sizeof(const char*)));
233 bool* cEntriesSelected = static_cast<bool*>(malloc(size*sizeof(bool)));
234 for (unsigned int i = 0; i < size; ++i)
235 {
236 cEntryIDs[i] = entries[i].id.c_str();
237 cEntryNames[i] = entries[i].name.c_str();
238 cEntriesSelected[i] = entries[i].selected;
239 }
240 bool ret = CAddonBase::m_interface->toKodi->kodi_gui->dialogSelect->open_multi_select(CAddonBase::m_interface->toKodi->kodiBase,
241 heading.c_str(), cEntryIDs, cEntryNames,
242 cEntriesSelected, size, autoclose);
243 if (ret)
244 {
245 for (unsigned int i = 0; i < size; ++i)
246 entries[i].selected = cEntriesSelected[i];
247 }
248 free(cEntryNames);
249 free(cEntryIDs);
250 free(cEntriesSelected);
251 return ret;
252 }
