504 inline static bool ADDON_GetDirectory(const AddonInstance_VFSEntry* instance,
505 const VFSURL* url,
506 VFSDirEntry** retEntries,
507 int* num_entries,
508 VFSGetDirectoryCallbacks* callbacks)
509 {
510 std::vector<kodi::vfs::CDirEntry> addonEntries;
511 bool ret = instance->toAddon.addonInstance->GetDirectory(*url, addonEntries, CVFSCallbacks(callbacks));
512 if (ret)
513 {
514 VFSDirEntry* entries = static_cast<VFSDirEntry*>(malloc(sizeof(VFSDirEntry) * addonEntries.size()));
515 for (unsigned int i = 0; i < addonEntries.size(); ++i)
516 {
517 entries[i].label = strdup(addonEntries[i].Label().c_str());
518 entries[i].title = strdup(addonEntries[i].Title().c_str());
519 entries[i].path = strdup(addonEntries[i].Path().c_str());
520 entries[i].folder = addonEntries[i].IsFolder();
521 entries[i].size = addonEntries[i].Size();
522
523 entries[i].num_props = 0;
524 const std::map<std::string, std::string>& props = addonEntries[i].GetProperties();
525 if (!props.empty())
526 {
527 entries[i].properties = static_cast<VFSProperty*>(malloc(sizeof(VFSProperty)*props.size()));
528 for (const auto& prop : props)
529 {
530 entries[i].properties[entries[i].num_props].name = strdup(prop.first.c_str());
531 entries[i].properties[entries[i].num_props].val = strdup(prop.second.c_str());
532 ++entries[i].num_props;
533 }
534 }
535 else
536 entries[i].properties = nullptr;
537 }
538 *retEntries = entries;
539 *num_entries = static_cast<int>(addonEntries.size());
540 }
541 return ret;
542 }
543
544 inline static void ADDON_FreeDirectory(const AddonInstance_VFSEntry* instance, VFSDirEntry* entries, int num_entries)
545 {
546 for (int i = 0; i < num_entries; ++i)
547 {
548 if (entries[i].properties)
549 {
550 for (unsigned int j = 0; j < entries[i].num_props; ++j)
551 {
552 free(entries[i].properties[j].name);
553 free(entries[i].properties[j].val);
554 }
555 free(entries[i].properties);
556 }
557 free(entries[i].label);
558 free(entries[i].title);
559 free(entries[i].path);
560 }
561 free(entries);
562 }
