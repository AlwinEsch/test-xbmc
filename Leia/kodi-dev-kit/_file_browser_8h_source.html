179 inline bool ShowAndGetFileList(const std::string& shares, const std::string& mask, const std::string& heading,
180 std::vector<std::string>& fileList, bool useThumbs = false, bool useFileDirectories = false)
181 {
182 using namespace ::kodi::addon;
183 char** list = nullptr;
184 unsigned int listSize = 0;
185 bool ret = CAddonBase::m_interface->toKodi->kodi_gui->dialogFileBrowser->show_and_get_file_list(CAddonBase::m_interface->toKodi->kodiBase,
186 shares.c_str(), mask.c_str(), heading.c_str(), &list, &listSize,
187 useThumbs, useFileDirectories);
188 if (ret)
189 {
190 for (unsigned int i = 0; i < listSize; ++i)
191 fileList.push_back(list[i]);
192 CAddonBase::m_interface->toKodi->kodi_gui->dialogFileBrowser->clear_file_list(CAddonBase::m_interface->toKodi->kodiBase, &list, listSize);
193 }
194 return ret;
195 }
